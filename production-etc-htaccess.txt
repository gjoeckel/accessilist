# .htaccess for /var/websites/webaim/htdocs/training/online/etc/
# Protects .env file from web access
# Deploy this file to: /var/websites/webaim/htdocs/training/online/etc/.htaccess

# Deny access to .env files
<Files ".env">
    Require all denied
</Files>

<Files ".env.*">
    Require all denied
</Files>

# Deny access to all files in this directory by default
<FilesMatch ".*">
    Require all denied
</FilesMatch>
